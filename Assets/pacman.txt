# Arch Pacman Package Manager Cheatsheet

# Update Whole System Packages
pacman -Syu

# Update Package Database
pacman -Syy

# Install Packages
pacman -S <pkg>

# Search Package
pacman -Ss <pkg>

# List Installed Packages
pacman -Q

# Search Installed Packages
pacman -Qs <pkg>

# List Explictly Installed Packages
pacman -Qe

# Package Information
pacman -Si <pkg>

# Installed Package Information
pacman -Qi <pkg>

# Download Package (/var/cache/pacman/pkg)
pacman -Sw <pkg>

# Install Downloaded Package
pacman -U <pkg>

# Remove Packages
pacman -R <pkg>

# Remove Package & Dependencies
pacman -Rs <pkg>

# Remove Packages & Configs, Dependencies
pacman -Rns <pkg>

# List Orphan/Obsolete Packages
pacman -Qtdq

# Remove Orphan/Obsolete Packages
pacman -Rns $(pacman -Qtdq)

# Remove Uninstalled Cached Packages
pacman -Sc

# Remove Cached Packages (All)
pacman -Scc
