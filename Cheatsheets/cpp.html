# Input & Output

• Output :

cout<< is used for printing output.
#include<iostream>
using namespace std;

int main(){
    cout << "Hello World!";

    return 0;
}

• Input :

cin>> is used for taking user input.
#include<iostream>
using namespace std;

int main(){
    int n;
    cout << "Enter Number:";
    cin >> n;

    return 0;
}

# Comments

• Single-line Comment :

// is used for single-line comment.
#include<iostream>
using namespace std;

int main(){
    cout << "Hello World!";

    // This is single-line comment.

    return 0;
}

• Multi-line Comment :

/* */ is used for multi-line comment.
#include<iostream>
using namespace std;

int main(){
    cout << "Hello World!";

    /* This is
    multi-line comment. */

    return 0;
}
